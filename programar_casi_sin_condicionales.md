```py
mapa_edificaciones = {
  "casa": Casa,
  "departamento": Departamento,
  "edificio": Edificio,
}

def obtener_edificacion(ed, id_, tipo, consumo, servicio):
  if tipo == 'oficina':
    t = Oficina()
  elif: tipo == 'domestico':
    t = Domestico()
  else:
    t = Comercial()
  
  c =  float(consumo)

  tipo_consumo = servicio.tipo_consumo()
  args = {"tipo":t, tipo_consumo:c}

  return mapa_edificaciones[ed](**args)
```
